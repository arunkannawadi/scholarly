103 def _get_page(self, pagerequest: str, premium: bool = False) -> str: 

104 """Return the data from a webpage 

105 

106 :param pagerequest: the page url 

107 :type pagerequest: str 

108 :param premium: whether or not to use the premium proxy right away 

109 :type premium: bool 

110 :returns: the text from a webpage 

111 :rtype: {str} 

112 :raises: MaxTriesExceededException, DOSException 

113 """ 

114 self.logger.info("Getting %s", pagerequest) 

115 resp = None 

116 tries = 0 

117 if ("citations?" in pagerequest) and (not premium): 

118 pm = self.pm2 

119 session = self._session2 

120 premium = False 

121 else: 

122 pm = self.pm1 

123 session = self._session1 

124 premium = True 

125 if pm._use_scraperapi: 

126 self.set_timeout(60) 

127 timeout=self._TIMEOUT 

128 while tries < self._max_retries: 

129 try: 

130 w = random.uniform(1,2) 

131 time.sleep(w) 

132 resp = session.get(pagerequest, timeout=timeout) 

133 self.logger.debug("Session proxy config is {}".format(session.proxies)) 

134 

135 has_captcha = self._requests_has_captcha(resp.text) 

136 

137 if resp.status_code == 200 and not has_captcha: 

138 return resp.text 

139 elif has_captcha: 

140 self.logger.info("Got a captcha request.") 

141 session = pm._handle_captcha2(pagerequest) 

142 continue # Retry request within same session 

143 elif resp.status_code == 403: 

144 self.logger.info("Got an access denied error (403).") 

145 if not pm.has_proxy(): 

146 self.logger.info("No other connections possible.") 

147 if not self.got_403: 

148 self.logger.info("Retrying immediately with another session.") 

149 else: 

150 if not pm._use_luminati: 

151 w = random.uniform(60, 2*60) 

152 self.logger.info("Will retry after {} seconds (with another session).".format(w)) 

153 time.sleep(w) 

154 self._new_session(premium=premium) 

155 self.got_403 = True 

156 

157 continue # Retry request within same session 

158 else: 

159 self.logger.info("We can use another connection... let's try that.") 

160 else: 

161 self.logger.info(f"""Response code {resp.status_code}. 

162 Retrying...""") 

163 

164 except DOSException: 

165 if not pm.has_proxy(): 

166 self.logger.info("No other connections possible.") 

167 w = random.uniform(60, 2*60) 

168 self.logger.info("Will retry after {} seconds (with the same session).".format(w)) 

169 time.sleep(w) 

170 continue 

171 except Timeout as e: 

172 err = f"Timeout Exception %s while fetching page: %s" % (type(e).__name__, e.args) 

173 self.logger.info(err) 

174 if timeout < 3*self._TIMEOUT: 

175 self.logger.info("Increasing timeout and retrying within same session.") 

176 timeout = timeout + self._TIMEOUT 

177 continue 

178 self.logger.info("Giving up this session.") 

179 except Exception as e: 

180 err = f"Exception %s while fetching page: %s" % (type(e).__name__, e.args) 

181 self.logger.info(err) 

182 self.logger.info("Retrying with a new session.") 

183 

184 tries += 1 

185 session, timeout = pm.get_next_proxy(num_tries = tries, old_timeout = timeout) 

186 

187 # If secondary proxy does not work, try again primary proxy. 

188 if not premium: 

189 return self._get_page(pagerequest, True) 

190 else: 

191 raise MaxTriesExceededException("Cannot Fetch from Google Scholar.")

225 def _has_captcha(self, got_id, got_class) -> bool: 

226 _CAPTCHA_IDS = [ 

227 "gs_captcha_ccl", # the normal captcha div 

228 "recaptcha", # the form used on full-page captchas 

229 "captcha-form", # another form used on full-page captchas
